async function fetchData() { try { console.log("正在请求数据..."); const res = await fetch('https://gl.asasasxzc.workers.dev'); if (!res.ok) { console.error("服务器响应异常:", res.status); return; } const data = await res.json(); console.log("收到的原始数据:", data); // 在浏览器 F12 就能看到数据 if (!data || data.length === 0) { console.warn("数据库里没数据，请先执行 INSERT！"); myChart.showLoading({ text: '数据库暂无数据' }); return; } myChart.hideLoading(); // 这里的映射逻辑要和你的 D1 字段名完全一致 const option = { backgroundColor: 'transparent', title: { text: '全球商贸脉动', left: 'center', textStyle: { color: '#00c853' } }, tooltip: { trigger: 'axis' }, xAxis: { type: 'category', data: data.map(i => i.timestamp.split('T')[1].slice(0, 5)) // 截取时间 12:00 }, yAxis: { type: 'value' }, series: [{ data: data.map(i => i.score), type: 'line', smooth: true, itemStyle: { color: '#00c853' }, areaStyle: { opacity: 0.2 } }] }; myChart.setOption(option); } catch (err) { console.error("JS 运行报错:", err); } }
